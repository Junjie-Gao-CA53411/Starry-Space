1. 使用方式：

# 标准检测
python hardware_profiler.py

# 详细日志（调试用）
python hardware_profiler.py -v

# 仅输出环境变量（供脚本调用）
python hardware_profiler.py --env

# 自定义输出路径
python hardware_profiler.py -o /path/to/profile.json


2. 集成到Portable Starry

# 启动器伪代码
import subprocess
import json

# 1. 运行硬件检测
subprocess.run(["python", "hardware_profiler.py", "-o", "data/profile.json"])

# 2. 读取配置
with open("data/profile.json") as f:
    profile = json.load(f)

# 3. 自动配置 Ollama
ollama_cfg = profile["ollama"]
for key, val in ollama_cfg["env_vars"].items():
    os.environ[key] = val

# 4. 启动 Ollama
subprocess.Popen(["ollama", "serve"])

# 5. 推荐用户下载模型
print(f"💡 推荐下载: ollama pull qwen2.5:{ollama_cfg['model_size']}")
